MySQL CheatSheet

Insert Update Delete

INSERT INTO 表名(属性1, 属性2, … , 属性m) VALUES(值1,值2, …, 值m); 
INSERT INTO 表名 [ (属性列表) ] VALUES(取值列表1),(取值列表2) … , (取值列表n);
INSERT INTO 表名1 (属性列表1) SELECT 属性列表2 FROM 表名2 WHERE 条件表达式;
UPDATE 表名 SET 属性名1=取值1, 属性名2=取值2, …, 属性 n=取值n WHERE 条件表达式;
DELETE FROM 表名 [WHERE 条件表式];

View

CREATE [ALGORITHM = {UNDEFINED|MERGE| TEMPTABLE}] VIEW 视图名 [(字段清单)] AS SELECT语句 [WITH [CASCADED|LOCAL] CHECK OPTION];
DESCRIBE 视图名; 
SHOW TABLE STATUS LIKE ‘视图名’; 
SHOW CREATE VIEW 视图名;
SELECT * FROM information_schema.views; 
CREATE OR REPLACE [ ALGORITHM = { UNDEFINED | MERGE | TEMPTABLE } ] VIEW 视图名 [ ( 属性清单 ) ] AS SELECT语句 [ WITH [ CASCADED | LOCAL ] CHECK OPTION]; 
ALTER [ ALGORITHM = { UNDEFINED | MERGE | TEMPTABLE } ] VIEW 视图名 [ ( 属性清单 ) ] AS SELECT语句 [ WITH [ CASCADED | LOCAL ] CHECK OPTION];
DROP VIEW [IF EXISTS] 视图名列表 [RESTRICT | CASCADE]

Trigger

CREATE TRIGGER 触发器名 BEFORE|AFTER 触发事件   ON 表名 FOR EACH ROW 执行语句
CREATE TRIGGER 触发器名 BEFORE | AFTER 触发事件 ON 表名 FOR EACH ROW BEGIN 执行语句列表 END 
DROP TRIGGER 触发器名;
